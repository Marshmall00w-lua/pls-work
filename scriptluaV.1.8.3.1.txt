local function e(f, g, i)
    local j = ""
    for k = 1, #f do
        local l = (k - 1) / (#f - 1)
        local n = g.R + (i.R - g.R) * l
        local o = g.G + (i.G - g.G) * l
        local p = g.B + (i.B - g.B) * l
        local q = string.format("#%02X%02X%02X", n * 255, o * 255, p * 255)
        j = j .. string.format('<font color="%s">%s</font>', q, f:sub(k, k))
    end
    return j
end

-- robust sapling finder
function findSapling()
    if workspace.Items then
        s = workspace.Items:FindFirstChild("Sapling")
        if s then return s end
        for _, child in ipairs(workspace.Items:GetChildren()) do
            if tostring(child.Name):lower():find("sapling") then return child end
        end
    end
    return nil
end

-- ground raycast under origin (start a bit above so we don't hit the player)
function getGroundPositionBelow(origin, ignoreList)
    rayOrigin = origin + Vector3.new(0, 3, 0)
    rayDir = Vector3.new(0, -300, 0)
    params = RaycastParams.new()
    params.FilterDescendantsInstances = ignoreList or {}
    params.FilterType = Enum.RaycastFilterType.Blacklist
    res = workspace:Raycast(rayOrigin, rayDir, params)
    if res and res.Position then
        return res.Position + Vector3.new(0, 0.06, 0) -- small offset, tweak if needed
    end
    return nil
end

-- defensive plant: try Vector3, then CFrame
function tryPlant(remote, sapling, pos)
    if not remote then return end
    pcall(function() remote:InvokeServer(sapling, pos) end)
    pcall(function() remote:InvokeServer(sapling, CFrame.new(pos)) end)
end

-- start single-safe planting loop
function startPlantingLoop()
    if plantingTaskRunning then return end
    plantingTaskRunning = true

    task.spawn(function()
        player = game.Players.LocalPlayer
        while plantingTaskRunning do
            if FollowPlayer then
                char = player and player.Character
                hrp = char and char:FindFirstChild("HumanoidRootPart")
                sapling = findSapling()
                if hrp and sapling then
                    -- plant SaplingAmount, but recalc position for each sapling so it follows movement
                    for i = 1, SaplingAmount do
                        if not FollowPlayer then break end

                        -- ensure server thinks we're dragging the sapling (do it each time)
                        pcall(function()
                            if RemoteStartDrag then RemoteStartDrag:FireServer(sapling) end
                        end)

                        -- recompute ground under current HRP so planting follows the player
                        ground = getGroundPositionBelow(hrp.Position, {char})
                        plantPos = ground or (hrp.Position + Vector3.new(0, -3, 0))

                        tryPlant(PlantRemote, sapling, plantPos)

                        task.wait(plantingDelay)
                    end
                else
                    -- no hrp or sapling, retry quickly
                    task.wait(0.12)
                end
            else
                -- not following, sleep and let toggle handle re-start
                task.wait(0.12)
            end
        end

        plantingTaskRunning = false
    end)
end

-- üåê Skid Fling Function
-- Replace your SkidFling with this version that respects flingBlocked
function SkidFling(Target)
    -- If flings are globally blocked, don't start
    if flingBlocked then return end

    -- whitelist check
    if table.find(whitelist, Target.Name) then
        return
    end

    root = Player.Character and Player.Character:FindFirstChild("HumanoidRootPart")
    tRoot = Target.Character and Target.Character:FindFirstChild("HumanoidRootPart")
    if not (root and tRoot) then return end

    old = root.CFrame
    bv = Instance.new("BodyVelocity", root)
    bv.Velocity = Vector3.new(9e8, 9e8, 9e8)
    bv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)

    t0 = tick()
    while tick() - t0 < 1.5 do
        -- if an interrupt was requested, stop the fling early
        if flingBlocked then
            break
        end

        if root and tRoot then
            root.CFrame = tRoot.CFrame
        end
        RunService.Heartbeat:Wait()
    end

    -- destroy velocity and only restore old position if not interrupted
    if bv and bv.Parent then pcall(function() bv:Destroy() end) end
    if not flingBlocked and old then
        pcall(function() root.CFrame = old end)
    end
end

    Callback = function()
        local char = Player.Character
        if not char then return end
        local hrp = char:FindFirstChild("HumanoidRootPart")
        local hum = char:FindFirstChildOfClass("Humanoid")
        if not hrp or not hum then return end

        -- Temporarily disable fling and block new ones
        local oldFlingState = flingAllEnabled
        flingAllEnabled = false
        flingBlocked = true

        task.wait(0.05) -- allow any active SkidFling to exit

        -- Clear existing momentum
        for _, child in ipairs(hrp:GetChildren()) do
            if child:IsA("BodyVelocity") or child:IsA("BodyGyro") or child:IsA("BodyPosition") or child:IsA("BodyAngularVelocity") then
                pcall(function() child:Destroy() end)
            end
        end
        -- Also reset velocity directly
        hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
        hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)

        -- Teleport to target location
        hrp.CFrame = CFrame.new(Vector3.new(4.95910645e-05, 9.5, -7.62939453e-06))

        -- Freeze character
        hum.WalkSpeed = 0
        hum.JumpPower = 0
        hrp.Anchored = true

        -- After 3 seconds, unfreeze and reset movement
        task.spawn(function()
            task.wait(3)
            hrp.Anchored = false
            hum.WalkSpeed = 16
            hum.JumpPower = 50

            -- Restore fling toggle
            flingBlocked = false
            flingAllEnabled = oldFlingState
        end)
    end
})

function invokeToolDamage(G, H, damageID, I)
    if client_ok and Client and Client.Events and Client.Events.ToolDamageObject then
        local ok, res =
            pcall(
            function()
                return Client.Events.ToolDamageObject:InvokeServer(G, H, damageID, I)
            end
        )
        if ok then
            return res
        end
    end
    fallback =
        ReplicatedStorage:FindFirstChild("RemoteEvents") and
        ReplicatedStorage.RemoteEvents:FindFirstChild("ToolDamageObject")
    if fallback and fallback:IsA("RemoteFunction") then
        ok, res =
            pcall(
            function()
                return fallback:InvokeServer(G, H, damageID, I)
            end
        )
        if ok then
            return res
        end
    end
    return nil
end

            local S = bE.X + math.cos(R) * distance
            local T = bE.Z + math.sin(R) * distance
            table.insert(bF, Vector3.new(S, bE.Y, T))
        end
    else
        local bG = math.ceil(math.sqrt(bD))
        local bH = (bG - 1) / 2

        for bJ = 0, bG - 1 do
            for bK = 0, bG - 1 do

                if bJ == 0 or bJ == bG - 1 or bK == 0 or bK == bG - 1 then
                    local S = bE.X + (bK - bH) * distance
                    local T = bE.Z + (bJ - bH) * distance
                    table.insert(bF, Vector3.new(S, bE.Y, T))

                c7 = {dI, dH, tostring(dP), d5.Character and d5.Character:GetPivot() or CFrame.new(part.Position)}
            end

            task.wait(0.12)
            local dR = nil
            local dS = dA(dI:GetAttribute("Health"), dI)
            if dS == true then
                hitCounter = hitCounter + 1
                return true, "health decreased with sig " .. dO .. " dmg=" .. tostring(dP)
            end
            local pos = part and part.Position and part.Position or nil
            local dT, dU = dv(pos, 0.25, 12)
            if dT then
                hitCounter = hitCounter + 1
                return true, "drop detected with sig " .. dO .. " dmg=" .. tostring(dP) .. " drop=" .. tostring(dU.Name)
            end
            if ok then
                hitCounter = hitCounter + 1
                return true, "remote call returned ok for sig " .. dO .. " dmg=" .. tostring(dP)
            end
            task.wait(0.2)
        end
    end
    return false, "exhausted signatures"
end

-- Cleanup old connections
_G.SafeSaveConn = _G.SafeSaveConn or nil

function CleanupConns()
    if _G.SafeSaveConn then pcall(function() _G.SafeSaveConn:Disconnect() end) _G.SafeSaveConn = nil end
    if _G.VoidCheckConn then pcall(function() _G.VoidCheckConn:Disconnect() end) _G.VoidCheckConn = nil end
end

-- Teleport helper
function TeleportToSafe()
    if not HRP or not _G.SafePosition then return end
    for i = 1, _G.TeleportAttempts do
        pcall(function()
            HRP.CFrame = CFrame.new(_G.SafePosition)
            if HRP.AssemblyLinearVelocity then HRP.AssemblyLinearVelocity = Vector3.new(0,0,0) end
            if HRP.AssemblyAngularVelocity then HRP.AssemblyAngularVelocity = Vector3.new(0,0,0) end
        end)
        wait(_G.TeleportAttemptDelay)
    end
end

-- Setup character and connections
function SetupCharacter(char)
    CleanupConns()
    Character = char or Player.Character
    if not Character then return end
    HRP = Character:FindFirstChild("HumanoidRootPart") or Character:WaitForChild("HumanoidRootPart")
    Humanoid = Character:FindFirstChild("Humanoid") or Character:WaitForChild("Humanoid")

    -- Save last safe position near Y‚âà6
    _G.SafeSaveConn = RunService.Heartbeat:Connect(function()
        if not _G.AntiVoidEnabled then return end
        if not Character or not Character.Parent then return end
        if not HRP then return end

        local curY = HRP.Position.Y
        if math.abs(curY - 6) <= _G.YTolerance then
            _G.SafePosition = Vector3.new(HRP.Position.X, 6, HRP.Position.Z)
        end
    end)

    -- Detect void: Y between -20 and -150
    _G.VoidCheckConn = RunService.Heartbeat:Connect(function()
        if not _G.AntiVoidEnabled then return end
        if not Character or not Character.Parent then return end
        if not HRP then return end
        if _G.AntiVoidDebounce then return end

        local curY = HRP.Position.Y
        if curY <= -20 and curY >= -150 then
            _G.AntiVoidDebounce = true
            pcall(TeleportToSafe)
            delay(0.5, function() _G.AntiVoidDebounce = false end)
        end
    end)
end

function MAPLOADER_GrassScan()
    if MAPLOADER_ACTIVE then
        return
    end
    MAPLOADER_ACTIVE = true
    d5 = Players.LocalPlayer
    if not d5 then
        return
    end
    char = d5.Character or d5.CharacterAdded:Wait()
    hrp = char:WaitForChild("HumanoidRootPart")
    grassFolder = Workspace:FindFirstChild("Map") and Workspace.Map:FindFirstChild("Ground")
    if not grassFolder then
        NOTIFY("GrassLoader", "No Map.Ground folder found!", 3)
        MAPLOADER_ACTIVE = false
        return
    end
    grassParts = {}
    for k, cO in pairs(grassFolder:GetChildren()) do
        if cO:IsA("BasePart") then
            table.insert(grassParts, cO)
        end
    end
    if #grassParts == 0 then
        NOTIFY("GrassLoader", "No grass parts found!", 3)
        MAPLOADER_ACTIVE = false
        return
    end
    NOTIFY("MapLoader", "Starting teleport (Delay: " .. LOAD_DELAY .. "s)", 2)
    for k = 1, #grassParts do
        if not MAPLOADER_ACTIVE then
            break
        end
        part = grassParts[k]
        if part and part.Parent then
            hrp.CFrame = CFrame.new(part.Position + Vector3.new(0, 3, 0))
        end
        task.wait(LOAD_DELAY)
    end
    MAPLOADER_ACTIVE = false
    NOTIFY("Map Loader", "Finished teleporting!", 3)
end

                AFK_Title.Text = "RyZen Hub AFK"
            end
        end
    )
    AFK_CONN_INPUT_BEGAN =
        AFK_Frame.InputBegan:Connect(
        function(ai)
            if ai.UserInputType == Enum.UserInputType.MouseButton1 or ai.UserInputType == Enum.UserInputType.Touch then
                AFK_DRAGGING = true
                AFK_DRAG_INPUT = ai
                if ai.UserInputType == Enum.UserInputType.MouseButton1 then
                    pos = UserInputService:GetMouseLocation()
                else
                    pos = ai.Position
                end
                AFK_DRAG_OFFSET_X = pos.X - AFK_Frame.AbsolutePosition.X
                AFK_DRAG_OFFSET_Y = pos.Y - AFK_Frame.AbsolutePosition.Y
            end
        end
    )
    AFK_CONN_INPUT_CHANGED =
        UserInputService.InputChanged:Connect(
        function(ai)
            if AFK_DRAGGING and AFK_DRAG_INPUT then
                if
                    ai == AFK_DRAG_INPUT or ai.UserInputType == Enum.UserInputType.MouseMovement or
                        ai.UserInputType == Enum.UserInputType.Touch
                 then
                    if
                        ai.UserInputType == Enum.UserInputType.MouseMovement or
                            ai.UserInputType == Enum.UserInputType.MouseButton1
                     then
                        pos = UserInputService:GetMouseLocation()
                    else
                        pos = ai.Position
                    end
                    newX = pos.X - AFK_DRAG_OFFSET_X
                    newY = pos.Y - AFK_DRAG_OFFSET_Y
                    cam = Workspace.CurrentCamera
                    if cam then
                        vs = cam.ViewportSize
                        if newX < 0 then
                            newX = 0
                        end
                        if newY < 0 then
                            newY = 0
                        end
                        maxX = vs.X - AFK_Frame.AbsoluteSize.X
                        maxY = vs.Y - AFK_Frame.AbsoluteSize.Y
                        if newX > maxX then
                            newX = maxX
                        end
                        if newY > maxY then
                            newY = maxY
                        end
                    end
                    AFK_Frame.Position = UDim2.new(0, newX, 0, newY)
                end
            end
        end
    )
    AFK_CONN_INPUT_ENDED =
        AFK_Frame.InputEnded:Connect(
        function(ai)
            if AFK_DRAG_INPUT and ai == AFK_DRAG_INPUT then
                AFK_DRAGGING = false
                AFK_DRAG_INPUT = nil
            end
        end
    )
    AFK_GUI.Enabled = true
end

-- Notification helper (detailed)
 function Notify(title, text, duration)
    duration = duration or 3
    if WindUI and type(WindUI.Notify) == "function" then
        pcall(function()
            WindUI:Notify({
                Title = title,
                Content = text,
                Duration = duration,
                Icon = "rbxassetid://84501312005643",
            })
        end)
    else
        StarterGui:SetCore("SendNotification", {
            Title = title,
            Text = text,
            Duration = duration
        })
    end
end

-- Coordinate GUI
 coordGui = nil
 coordLabel = nil
 coordConnection = nil

-- Toggle coordinates display
 function ToggleCoordinates(enabled)
    if type(enabled) == "boolean" then
        ShowCoordinates = enabled
    else
        ShowCoordinates = not ShowCoordinates
    end

    if ShowCoordinates then
        if not coordGui then
            coordGui = Instance.new("ScreenGui")
            coordGui.Name = "CoordsGUI"
            coordGui.ResetOnSpawn = false
            coordGui.Parent = game.CoreGui

            coordLabel = Instance.new("TextLabel")
            coordLabel.Size = UDim2.new(0, 220, 0, 36)
            coordLabel.Position = UDim2.new(0, 10, 0, 10)
            coordLabel.BackgroundTransparency = 0.4
            coordLabel.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
            coordLabel.BorderSizePixel = 0
            coordLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
            coordLabel.TextScaled = true
            coordLabel.Font = Enum.Font.SourceSansSemibold
            coordLabel.Text = "X: 0.0  Y: 0.0  Z: 0.0"
            coordLabel.Parent = coordGui
        end

        if not coordConnection then
            coordConnection = RunService.RenderStepped:Connect(function()
                 char = LocalPlayer.Character
                 root = char and char:FindFirstChild("HumanoidRootPart")
                if root then
                     p = root.Position
                    CurrentCoordinates = p
                    coordLabel.Text = string.format("X: %.2f  Y: %.2f  Z: %.2f", p.X, p.Y, p.Z)
                    coordLabel.Visible = true
                else
                    coordLabel.Text = "No character"
                end
            end)
        end
    else
        if coordConnection then
            coordConnection:Disconnect()
            coordConnection = nil
        end
        if coordGui then
            coordGui:Destroy()
            coordGui = nil
            coordLabel = nil
        end
    end
end

-- Refresh dropdown
 function RefreshDropdownValues(dropdown)
    if not dropdown then return end
     keys = {}
    for k in pairs(SavedPositions) do table.insert(keys, k) end
    table.sort(keys)
    dropdown:Refresh(keys)
end

-- Save position
 function SavePosition(name, dropdown)
    name = tostring(name or ""):gsub("^%s*(.-)%s*$", "%1")
    if name == "" then
        return Notify("Error", "Please enter a name for the coordinate!", 3)
    end

     char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
     root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then
        return Notify("Error", "HumanoidRootPart not found!", 3)
    end

     pos = root.Position
    SavedPositions[name] = root.CFrame
    Notify("Saved", string.format("'%s' saved at X: %.2f Y: %.2f Z: %.2f", name, pos.X, pos.Y, pos.Z), 3)
    RefreshDropdownValues(dropdown)
end

-- Teleport
 function TeleportToSaved()
    if not SelectedPosition or not SavedPositions[SelectedPosition] then
        return Notify("Error", "Select a saved position first!", 3)
    end

     char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
     root = char and char:FindFirstChild("HumanoidRootPart")
    if root then
        root.CFrame = SavedPositions[SelectedPosition]
         pos = root.Position
        Notify("Teleported", string.format("Teleported to '%s' at X: %.2f Y: %.2f Z: %.2f", SelectedPosition, pos.X, pos.Y, pos.Z), 3)
    else
        Notify("Error", "Couldn't find HumanoidRootPart.", 3)
    end
end

-- Delete
 function DeleteSaved(dropdown)
    if not SelectedPosition or not SavedPositions[SelectedPosition] then
        return Notify("Error", "Select a saved position first!", 3)
    end

    SavedPositions[SelectedPosition] = nil
    Notify("Deleted", "Deleted coordinate '" .. SelectedPosition .. "'", 2)
    SelectedPosition = nil
    RefreshDropdownValues(dropdown)
end

-- Copy current coordinates
 function CopyCoordinates()
     char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
     root = char and char:FindFirstChild("HumanoidRootPart")
    if root then
         pos = root.Position
         coordText = string.format("X: %.2f, Y: %.2f, Z: %.2f", pos.X, pos.Y, pos.Z)
        -- Roblox clipboard
        pcall(function() setclipboard(coordText) end)
        Notify("Copied", "Copied current coordinates to clipboard: " .. coordText, 3)
    else
        Notify("Error", "Couldn't find HumanoidRootPart.", 3)
    end
end
